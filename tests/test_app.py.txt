import pytest
from app import app

@pytest.fixture
def client():
    with app.test_client() as client:
        yield client

def test_home(client):
    response = client.get('/')
    assert response.status_code == 200
    assert response.json == {"message": "Welcome to the Python CI/CD Pipeline Template API!"}

def test_status(client):
    response = client.get('/status')
    assert response.status_code == 200
    assert response.json["status"] == "Running"
